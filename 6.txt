DECLARE

Declare variables to hold the data fetched from the cursor

v_roll_no oldrecords.roll_no%TYPE;

v_name oldrecords.name%TYPE;

Declare a cursor to select records from oldrecords that are not in newrecords

CURSOR c1 IS
SELECT roll_no, name FROM oldrecords

WHERE roll_no NOT IN (SELECT roll_no FROM newrecords);

BEGIN

Open the cursor OPEN C1;

Loop through the records fetched by the cursor

LOOP

FETCH c1 INTO v_roll_no, v_name;

Exit the loop when there are no more records to fetch EXIT WHEN C1% NOT FOUND;

Insert the fetched record into the newrecords table INSERT INTO newrecords (roll_no, name) VALUES (v_roll_no, v_name);

Output the inserted record for verification

DBMS_OUTPUT.PUT_LINE(‘Inserted Roll No: || v_roll_no ||, Name: || v_name);

END LOOP:

DBMS_OUTPUT.PUT_LINE(‘Inserted Roll No: || v_roll_no || ‘, Name: || v_name); END LOOP;

Close the cursor

CLOSE C1;

END;
